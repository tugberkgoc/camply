\subsection*{Install}


\begin{DoxyCode}
$ npm install decompress-tar
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
const decompress = require('decompress');
const decompressTar = require('decompress-tar');

decompress('unicorn.tar', 'dist', \{
  plugins: [
    decompressTar()
  ]
\}).then(() => \{
  console.log('Files decompressed');
\});
\end{DoxyCode}
