\subsection*{Usage}


\begin{DoxyCode}
const decompress = require('decompress');
const decompressUnzip = require('decompress-unzip');

decompress('unicorn.zip', 'dist', \{
  plugins: [
    decompressUnzip()
  ]
\}).then(() => \{
  console.log('Files decompressed');
\});
\end{DoxyCode}
