\subsection*{Usage}


\begin{DoxyCode}
const decompress = require('decompress');
const decompressTarbz = require('decompress-tarbz2');

decompress('unicorn.tar.gz', 'dist', \{
  plugins: [
    decompressTarbz()
  ]
\}).then(() => \{
  console.log('Files decompressed');
\});
\end{DoxyCode}
